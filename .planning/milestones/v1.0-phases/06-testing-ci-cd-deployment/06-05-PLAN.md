    ```typescript
    import { expect, test } from "@playwright/test";

    const API_URL = process.env.PLAYWRIGHT_API_URL ?? "http://localhost:3001";

    // TST-04: DLQ dashboard re-queue flow against the live docker compose stack.
    // Precondition: the CI job seeds >=1 dlq_events row (mock-crm fail mode + fired signed webhook).
    // Exactly-once persistence is proven by TST-03 (IDM-02); this test proves the UI recovery path.
    test("DLQ re-queue: click Re-queue Job, see success, entry leaves the queue", async ({ page }) => {
      // 1. Load /dlq and wait for the seeded entry to render.
      await page.goto("/dlq");
      const firstRow = page.locator("table tbody tr").first();
      await expect(firstRow).toBeVisible({ timeout: 30_000 });

      // 2. Capture the fingerprint of the entry we re-queue (for the exactly-once-via-resolved check).
      const before = await page.request.get(`${API_URL}/api/dlq`);
      const beforeBody = (await before.json()) as { entries: Array<{ fingerprint: string }> };
      expect(beforeBody.entries.length).toBeGreaterThanOrEqual(1);
      const fingerprint = beforeBody.entries[0].fingerprint;

      // 3. Click the first "Re-queue Job" button.
      await page.getByRole("button", { name: "Re-queue Job" }).first().click();

      // 4. Assert the success feedback appears.
      await expect(page.getByText("Re-queued successfully.")).toBeVisible({ timeout: 15_000 });

      // 5. Assert the entry leaves the unresolved DLQ list (requeueDlqEntry set resolved=true).
      //    This proves the UI-triggered recovery completed end-to-end. Re-queue is idempotent by
      //    construction (jobId=fingerprint + ON CONFLICT DO NOTHING -> exactly one events row, per TST-03).
      await expect
        .poll(
          async () => {
            const res = await page.request.get(`${API_URL}/api/dlq`);
            const body = (await res.json()) as { entries: Array<{ fingerprint: string }> };
            return body.entries.some((e) => e.fingerprint === fingerprint);
          },
          { timeout: 20_000, intervals: [1000] },
        )
        .toBe(false);
    });
    ```

    ```yaml
      e2e:
        name: Playwright E2E (DLQ re-queue)
        runs-on: ubuntu-latest
        needs: verify
        steps:
          - uses: actions/checkout@v4

          - uses: pnpm/action-setup@v4

          - uses: actions/setup-node@v4
            with:
              node-version: 22
              cache: pnpm

          - name: Install dependencies
            run: pnpm install --frozen-lockfile

          - name: Install Playwright browser
            run: pnpm exec playwright install --with-deps chromium

          - name: Create .env for compose
            run: cp .env.example .env

          - name: Start compose stack
            run: docker compose up --build -d

          - name: Wait for API + dashboard health
            run: |
              timeout 180 bash -c 'until curl -sf http://localhost:3001/healthz >/dev/null; do sleep 2; done'
              timeout 180 bash -c 'until curl -sf http://localhost:3000 >/dev/null; do sleep 2; done'

          - name: Seed a DLQ entry (mock-crm fail mode + signed webhook)
            run: |
              set -euo pipefail
              # 1. Force mock-crm to fail every CRM sync.
              curl -sf -X POST http://localhost:3002/admin/failure-mode \
                -H 'content-type: application/json' -d '{"mode":"fail","rate":1}'
              # 2. Fire one valid signed shopee webhook (HMAC over the exact body bytes).
              SECRET="dev-secret-shopee"
              BODY='{"eventType":"order.created","externalId":"e2e-seed-1","occurredAt":"2026-01-01T00:00:00.000Z","payload":{"amount":1}}'
              SIG=$(printf '%s' "$BODY" | openssl dgst -sha256 -hmac "$SECRET" | sed 's/^.* //')
              curl -sf -X POST http://localhost:3001/ingest/shopee \
                -H 'content-type: application/json' \
                -H "x-webhook-signature: $SIG" \
                -d "$BODY"

          - name: Wait for DLQ entry to appear (retries exhaust -> dlq_events)
            run: |
              timeout 120 bash -c 'until [ "$(curl -sf http://localhost:3001/api/dlq | node -e "let d=\"\";process.stdin.on(\"data\",c=>d+=c).on(\"end\",()=>console.log(JSON.parse(d).entries.length))")" -ge 1 ]; do sleep 3; done'

          - name: Run Playwright E2E
            run: pnpm exec playwright test
            env:
              PLAYWRIGHT_BASE_URL: http://localhost:3000
              PLAYWRIGHT_API_URL: http://localhost:3001

          - name: Collect compose logs on failure
            if: failure()
            run: docker compose logs --no-color > compose-logs.txt || true

          - uses: actions/upload-artifact@v4
            if: always()
            with:
              name: e2e-artifacts
              path: |
                playwright-report/
                test-results/
                compose-logs.txt

          - name: Teardown
            if: always()
            run: docker compose down -v
    ```

    NOTE on the seed signing: `dev-secret-shopee` MUST match WEBHOOK_SECRET_SHOPEE in .env.example (seeded by Plan 06-01). The HMAC must be computed over the EXACT body bytes sent (use `printf '%s'` — no trailing newline). The `openssl dgst -hmac` output format is `(stdin)= <hex>`; the `sed 's/^.* //'` strips the prefix to leave the bare hex.

    Add `"test:e2e": "playwright test"` to the root package.json scripts (optional convenience; do NOT change other scripts). Validate the YAML indentation (e2e job is a sibling under jobs:).
